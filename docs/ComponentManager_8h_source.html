29 class ConcreteComponentContainer : public ComponentContainer {
30  using ListIter = typename std::list<T>::iterator;
31  struct ListIterHash {
32  size_t operator()(const ListIter &i) const {
33  return std::hash<T *>()(&*i);
34  }
35  };
36  std::list<T> components;
37  std::unordered_map<EntityID, ListIter> entityComponentMapping;
38  std::unordered_map<ListIter, EntityID, ListIterHash> componentEntityMapping;
39  public:
40  void insertEntityComponent(EntityID entity, T component) {
41  if (entityComponentMapping.count(entity)) throw ComponentAlreadyExist{};
42  ListIter iter = components.insert(components.end(), std::move(component));
43  entityComponentMapping[entity] = iter;
44  componentEntityMapping[iter] = entity;
45  }
46  void removeEntityComponent(EntityID entity) {
47  if (!entityComponentMapping.count(entity)) throw EntityDoesNotExist{};
48  auto iter = entityComponentMapping[entity];
49  components.erase(iter);
50  entityComponentMapping.erase(entity);
51  componentEntityMapping.erase(iter);
52  }
53  T &getComponent(EntityID entity) {
54  if (!entityComponentMapping.count(entity)) throw ComponentDoesNotExist{};
55  return *entityComponentMapping[entity];
56  }
57  bool hasComponent(EntityID entity) {
58  return entityComponentMapping.count(entity) > 0;
59  }
60  void entityDestroyed(EntityID entity) override {
61  if (entityComponentMapping.count(entity)) {
62  removeEntityComponent(entity);
63  }
64  }
65 };
